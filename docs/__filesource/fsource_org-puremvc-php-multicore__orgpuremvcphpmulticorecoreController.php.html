  22. require_once 'org/puremvc/php/multicore/interfaces/IController.php';
  23. require_once 'org/puremvc/php/multicore/interfaces/INotification.php';
  24. require_once 'org/puremvc/php/multicore/patterns/observer/Observer.php';
  25. require_once 'org/puremvc/php/multicore/core/View.php';
  26.  
  27. /**
  28.  * A Multiton <b>IController</b> implementation.
  29.  *
  30.  * In PureMVC, the <b>Controller</b> class follows the
  31.  * 'Command and Controller' strategy, and assumes these
  32.  * responsibilities:
  33.  *
  34.  * - Remembering which <b>ICommand</b>s  are intended to
  35.  *   handle which <b>INotifications</b>.
  36.  * - Registering itself as an <b>IObserver</b> with
  37.  *   the <b>View</b> for each <b>INotification</b>
  38.  *   that it has an <b>ICommand</b> mapping for.
  39.  * - Creating a new instance of the proper <b>ICommand</b>
  40.  *   to handle a given <b>INotification</b> when notified
  41.  *   by the <b>View</b>.
  42.  * - Calling the <b>ICommand</b>'s <b>execute</b> method,
  43.  *   passing in the <b>INotification</b>.
  44.  *
  45.  * Your application must register <b>ICommands</b> with the
  46.  * Controller.
  47.  *
  48.  * The simplest way is to subclass <b>Facade</b>,
  49.  * and use its <b>initializeController</b> method to add your
  50.  * registrations.
  51.  *
  52.  * @see MacroCommand
  53.         org\puremvc\php\multicore\patterns\command\MacroCommand.php
  54.  * @see SimpleCommand
  55.         org\puremvc\php\multicore\patterns\command\SimpleCommand.php
  56.  * @see Notification
  57.         org\puremvc\php\multicore\patterns\observer\Notification.php
  58.  * @see Observer
  59.         org\puremvc\php\multicore\patterns\observer\Observer.php
  60.  * @see View
  61.         org\puremvc\php\multicore\core\View.php
  62.  *
  63.  * @package org.puremvc.php.multicore
  64.  *
  65.  */
  66. class Controller implements IController
  67. {
  68.     /**
  69.      * Define the message content for the duplicate instance exception
  70.      * @var string 
  71.      */
  72.     const MULTITON_MSG = "Controller instance for this Multiton key already constructed!";
  73.  
  74.     /**
  75.      * The view instance for this Core
  76.      * @var IView 
  77.      */
  78.     protected $view;
  79.  
  80.     /**
  81.      * Mapping of Notification names to Command Class references
  82.      * @var array 
  83.      */
  84.     protected $commandMap;
  85.  
  86.     /**
  87.      * The Multiton Key for this Core
  88.      * @var string 
  89.      */
  90.     protected $multitonKey;
  91.  
  92.     /**
  93.      * The Multiton instances stack
  94.      * @var array 
  95.      */
  96.     protected static $instanceMap = array();
  97.  
  98.     /**
  99.      * Instance constructor
  100.      *
  101.      * This <b>IController</b> implementation is a Multiton,
  102.      * so you should not call the constructor
  103.      * directly, but instead call the static <b>getInstance() Factory</b> method,
  104.      * passing the unique key for this instance.
  105.      *
  106.      * ex:
  107.      * <code>
  108.      * $myController = MyController::getInstance( 'myMultitonKey' );
  109.      * </code>
  110.      *
  111.      * @param string $key Unique key for this instance.
  112.      * @throws Exception if instance for this key has already been constructed.
  113.      * @return IController Instance for this key
  114.      */
  115.     protected function __construct( $key )
  116.     {
  117.         if ( isset( self::$instanceMap[ $key ] ) )
  118.         {
  119.             throw new Exception(self::MULTITON_MSG);
  120.         }
  121.         $this->multitonKey = $key;
  122.         self::$instanceMap[ $this->multitonKey ] = $this;
  123.         $this->commandMap = array();
  124.         $this->initializeController();
  125.     }
  126.  
  127.     /**
  128.      * Initialize the instance.
  129.      *
  130.      * Called automatically by the constructor.
  131.      *
  132.      * Note that if you are using a subclass of <b>View</b> in your application,
  133.      * you should <i>also</i> subclass <b>Controller</b> and override the <i>initializeController()</i>
  134.      * method in the following way:
  135.      *
  136.      * <code>
  137.      * // ensure that the Controller is talking to my IView implementation
  138.      * public function initializeController( )
  139.      * {
  140.      *     $this->view = MyView::getInstance('myViewName');
  141.      * }
  142.      * </code>
  143.      *
  144.      * @return void 
  145.      */
  146.     protected function initializeController(  )
  147.     {
  148.         $this->view = View::getInstance( $this->multitonKey );
  149.     }
  150.  
  151.     /**
  152.      * Controller Factory method.
  153.      *
  154.      * This <b>IController</b> implementation is a Multiton so
  155.      * this method MUST be used to get acces, or create, <b>IController</b>s.
  156.      *
  157.      * @param string $key Unique key for this instance.
  158.      * @return IController The instance for this Multiton key.
  159.      */
  160.     public static function getInstance( $key )
  161.     {
  162.         if ( !isset( self::$instanceMap[ $key ] ) )
  163.         {
  164.             self::$instanceMap[ $key ] = new Controller( $key );
  165.         }
  166.  
  167.         return self::$instanceMap[ $key ];
  168.     }
  169.  
  170.     /**
  171.      * Execute Command
  172.      *
  173.      * Execute the <b>ICommand</b> previously registered as the
  174.      * handler for <b>INotification</b>s with the given notification name.
  175.      *
  176.      * @param INotification $notification The <b>INotification</b> to execute the associated <b>Command</b> for.
  177.      * @return void 
  178.      */
  179.     public function executeCommand( INotification $notification )
  180.     {
  181.         // if the Command is registered...
  182.         if( $this->hasCommand( $notification->getName() ) )
  183.         {
  184.             $commandClassName = $this->commandMap[ $notification->getName() ];
  185.             $commandClassRef = new $commandClassName();
  186.             $commandClassRef->initializeNotifier($this->multitonKey);
  187.             $commandClassRef->execute( $notification );
  188.         }
  189.     }
  190.  
  191.     /**
  192.      * Register Command
  193.      *
  194.      * Register a particular <b>ICommand</b> class as the handler
  195.      * for a particular <b>INotification</b>.
  196.      *
  197.      * If an <b>ICommand</b> has already been registered to
  198.      * handle <b>INotification</b>s with this name, it is no longer
  199.      * used, the new <b>ICommand</b> is used instead.
  200.      *
  201.      * The <b>IObserver</b> for the new <b>ICommand</b> is only created if this the
  202.      * first time an <b>ICommand</b>has been regisered for this <b>INotification</b> name.
  203.      *
  204.      * @param string $notificationName Name of the <b>INotification</b>.
  205.      * @param string $commandClassRef Class name of the <b>ICommand</b> implementation to register.
  206.      * @return void 
  207.      */
  208.     public function registerCommand( $notificationName, $commandClassRef )
  209.     {
  210.         if ( !$this->hasCommand( $notificationName ) )
  211.         {
  212.             $this->view->registerObserver( $notificationName, new Observer( "executeCommand", $this ) );
  213.         }
  214.         $this->commandMap[ $notificationName ] = $commandClassRef;
  215.     }
  216.  
  217.     /**
  218.      * Has Command
  219.      *
  220.      * Check if a <b>ICommand</b> is <b>registerCommand() registered</b> for a given <b>INotification</b>
  221.      *
  222.      * @param string $notificationName Name of the <b>INotification</b> to check for.
  223.      * @return bool Whether a <b>ICommand</b> is currently registered for the given <var>notificationName</var>.
  224.      */
  225.     public function hasCommand( $notificationName )
  226.     {
  227.         return isset( $this->commandMap[ $notificationName ] );
  228.     }
  229.  
  230.     /**
  231.      * Remove Command
  232.      *
  233.      * Remove a previously <b>registerCommand() registered</b> <b>ICommand</b> to <b>INotification</b> mapping.
  234.      *
  235.      * @param string $notificationName Name of the <b>INotification</b> to remove the <b>ICommand</b> mapping for.
  236.      * @return void 
  237.      */
  238.     public function removeCommand( $notificationName )
  239.     {
  240.         // if the Command is registered...
  241.         if ( $this->hasCommand( $notificationName ) )
  242.         {
  243.             // remove the observer
  244.             $this->view->removeObserver( $notificationName, $this );
  245.  
  246.             // remove the command
  247.             unset( $this->commandMap[ $notificationName ] );
  248.         }
  249.     }
  250.  
  251.     /**
  252.      * Remove controller
  253.      *
  254.      * Remove an <b>IController</b> instance identified by it's <b>key</b>
  255.      *
  256.      * @param string $key multitonKey of <b>IController</b> instance to remove
  257.      * @return void 
  258.      */
  259.     public static function removeController( $key )
  260.     {
  261.         unset( self::$instanceMap[ $key ] );
  262.     }
  263.  
  264. }
