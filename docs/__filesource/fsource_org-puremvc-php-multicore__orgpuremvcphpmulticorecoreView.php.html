  27. /**
  28.  * A Multiton <b>IView</b> implementation.
  29.  *
  30.  * In PureMVC, the <b>View</b> class assumes these responsibilities:
  31.  *
  32.  * - Maintain a cache of <b>IMediator</b> instances.
  33.  * - Provide methods for registering, retrieving, and removing <b>IMediators</b>.
  34.  * - Notifiying <b>IMediators</b> when they are registered or removed.
  35.  * - Managing the observer lists for each <b>INotification<b> in the application.
  36.  * - Providing a method for attaching <b>IObservers</b> to an <b>INotification</b>'s observer list.
  37.  * - Providing a method for broadcasting an <b>INotification</b>.
  38.  * - Notifying the <b>IObservers</b> of a given <b>INotification</b> when it broadcast.
  39.  *
  40.  * @see Mediator
  41.         org\puremvc\php\multicore\patterns\mediator\Mediator.php
  42.  * @see Observer
  43.         org\puremvc\php\multicore\patterns\observer\Observer.php
  44.  * @see Notification
  45.         org\puremvc\php\multicore\patterns\observer\Notification.php
  46.  *
  47.  * @package org.puremvc.php.multicore
  48.  */
  49. class View implements IView
  50. {
  51.     /**
  52.      * Define the message content for the duplicate instance exception
  53.      * @var string 
  54.      */
  55.     const MULTITON_MSG = "View instance for this Multiton key already constructed!";
  56.  
  57.     /**
  58.      * Mapping of mediatorNames to IMediator references
  59.      * @var array 
  60.      */
  61.     protected $mediatorMap = array();
  62.  
  63.     /**
  64.      * Mapping of Notification names to Observer lists
  65.      * @var array 
  66.      */
  67.     protected $observerMap  = array();
  68.  
  69.     /**
  70.      * The Multiton Key for this Core
  71.      * @var string 
  72.      */
  73.     protected $multitonKey = NULL;
  74.  
  75.     /**
  76.      * The Multiton instances stack
  77.      * @var array 
  78.      */
  79.     protected static $instanceMap = array();
  80.  
  81.     /**
  82.      * Constructor.
  83.      *
  84.      * This <b>IView</b> implementation is a Multiton,
  85.      * so you should not call the constructor
  86.      * directly, but instead call the static Multiton
  87.      * Factory method.
  88.      *
  89.      * <code>
  90.      * View::getInstance( 'multitonKey' );
  91.      * </code>
  92.      *
  93.      * @param string $key Unique key for this instance.
  94.      * @throws Exception if instance for this key has already been constructed.
  95.      * @return IView Instance for this key
  96.      */
  97.     protected function __construct( $key )
  98.     {
  99.         if ( isset( self::$instanceMap[ $key ] ) )
  100.         {
  101.             throw new Exception(self::MULTITON_MSG);
  102.         }
  103.         $this->multitonKey = $key;
  104.         self::$instanceMap[ $this->multitonKey ] = $this;
  105.         $this->mediatorMap = array();
  106.         $this->observerMap = array();
  107.         $this->initializeView();
  108.     }
  109.  
  110.     /**
  111.      * Initialize the Singleton View instance.
  112.      *
  113.      * Called automatically by the constructor, this
  114.      * is your opportunity to initialize the Multiton
  115.      * instance in your subclass without overriding the
  116.      * constructor.
  117.      *
  118.      * @return void 
  119.      */
  120.     protected function initializeView(  )
  121.     {
  122.     }
  123.  
  124.     /**
  125.      * View Factory method.
  126.      *
  127.      * This <b>IView</b> implementation is a Multiton so
  128.      * this method MUST be used to get acces, or create, <b>IView</b>s.
  129.      *
  130.      * @param string $key Unique key for this instance.
  131.      * @return IView The instance for this Multiton key.
  132.      */
  133.     public static function getInstance( $key )
  134.     {
  135.         if ( !isset( self::$instanceMap[ $key ] ) )
  136.         {
  137.             self::$instanceMap[ $key ] = new View( $key );
  138.         }
  139.         return self::$instanceMap[ $key ];
  140.     }
  141.  
  142.     /**
  143.      * Register Observer
  144.      *
  145.      * Register an <b>IObserver</b> to be notified
  146.      * of <b>INotifications</b> with a given name.
  147.      *
  148.      * @param string $notificationName The name of the <b>INotifications</b> to notify this <b>IObserver</b> of.
  149.      * @param IObserver $observer The <b>IObserver</b> to register.
  150.      * @return void 
  151.      */
  152.     public function registerObserver ( $notificationName, IObserver $observer)
  153.     {
  154.         if( isset( $this->observerMap[ $notificationName ] ) )
  155.         {
  156.             array_push($this->observerMap[ $notificationName ], $observer );
  157.         }
  158.         else
  159.         {
  160.             $this->observerMap[ $notificationName ] = array( $observer );
  161.         }
  162.     }
  163.  
  164.  
  165.     /**
  166.      * Notify Observers
  167.      *
  168.      * Notify the <b>IObservers</b> for a particular <b>INotification</b>.
  169.      *
  170.      * All previously attached <b>IObservers</b> for this <b>INotification</b>'s
  171.      * list are notified and are passed a reference to the <b>INotification</b> in
  172.      * the order in which they were registered.
  173.      *
  174.      * @param INotification $notification The <b>INotification</b> to notify <b>IObservers</b> of.
  175.      * @return void 
  176.      */
  177.     public function notifyObservers( INotification $notification )
  178.     {
  179.         if( isset( $this->observerMap[ $notification->getName() ] ) )
  180.         {
  181.             // Get a reference to the observers list for this notification name
  182.             $observers_ref = $this->observerMap[ $notification->getName() ];
  183.  
  184.             // Copy observers from reference array to working array,
  185.             // since the reference array may change during the notification loop
  186.             $observers = array();
  187.             foreach($observers_ref as $observer)
  188.             {
  189.                 array_push( $observers, $observer );
  190.             }
  191.  
  192.             // Notify Observers from the working array
  193.             foreach($observers as $observer)
  194.             {
  195.                 $observer->notifyObserver( $notification );
  196.             }
  197.         }
  198.     }
  199.  
  200.     /**
  201.      * Remove Observer
  202.      *
  203.      * Remove a group of observers from the observer list for a given Notification name.
  204.      *
  205.      * @param string $notificationName Which observer list to remove from.
  206.      * @param mixed $notifyContext Remove the observers with this object as their notifyContext
  207.      * @return void 
  208.      */
  209.     public function removeObserver( $notificationName, $notifyContext )
  210.     {
  211.         //Is there registered Observers for the notification under inspection
  212.         if( !isset( $this->observerMap[ $notificationName ] )) return;
  213.  
  214.         // the observer list for the notification under inspection
  215.         $observers = $this->observerMap[ $notificationName ];
  216.  
  217.         // find the observer for the notifyContext
  218.         for ( $i = 0; $i < count( $observers ); $i++ )
  219.         {
  220.             if ( $observers[$i]->compareNotifyContext( $notifyContext ) == true )
  221.             {
  222.                 // there can only be one Observer for a given notifyContext
  223.                 // in any given Observer list, so remove it and break
  224.                 array_splice($observers,$i,1);
  225.                 break;
  226.             }
  227.         }
  228.  
  229.         // Also, when a Notification's Observer list length falls to
  230.         // zero, delete the notification key from the observer map
  231.         if ( count( $observers ) == 0 )
  232.         {
  233.             unset($this->observerMap[ $notificationName ]);
  234.         }
  235.     }
  236.  
  237.     /**
  238.      * Register Mediator
  239.      *
  240.      * Register an <b>IMediator</b> instance with the <b>View</b>.
  241.      *
  242.      * Registers the <b>IMediator</b> so that it can be retrieved by name,
  243.      * and further interrogates the <b>IMediator</b> for its
  244.      * <b>INotification</b> interests.
  245.      *
  246.      * If the <b>IMediator</b> returns any <b>INotification</b>
  247.      * names to be notified about, an <b>Observer</b> is created encapsulating
  248.      * the <b>IMediator</b> instance's <b>handleNotification</b> method
  249.      * and registering it as an <b>Observer</b> for all <b>INotifications</b> the
  250.      * <b>IMediator</b> is interested in.
  251.      *
  252.      * @param IMediator $mediator Reference to the <b>IMediator</b> instance.
  253.      * @return void 
  254.      */
  255.     public function registerMediator( IMediator $mediator )
  256.     {
  257.  
  258.         // do not allow re-registration (you must to removeMediator fist)
  259.         if ( $this->hasMediator( $mediator->getMediatorName() ) ) return;
  260.  
  261.         $mediator->initializeNotifier( $this->multitonKey );
  262.  
  263.         // Register the Mediator for retrieval by name
  264.         $this->mediatorMap[ $mediator->getMediatorName() ] = $mediator;
  265.  
  266.         // Get Notification interests, if any.
  267.         $interests = array();
  268.         $interests = $mediator->listNotificationInterests();
  269.  
  270.         // Register Mediator as an observer for each notification of interests
  271.         if ( count( $interests ) > 0 )
  272.         {
  273.             // Create Observer referencing this mediator's handlNotification method
  274.             $observer = new Observer( "handleNotification", $mediator );
  275.  
  276.             // Register Mediator as Observer for its list of Notification interests
  277.             for ( $i = 0;  $i < count( $interests ); $i++ )
  278.             {
  279.                 $this->registerObserver( $interests[$i],  $observer );
  280.             }
  281.         }
  282.  
  283.         // alert the mediator that it has been registered
  284.         $mediator->onRegister();
  285.  
  286.     }
  287.  
  288.     /**
  289.      * Retreive Mediator
  290.      *
  291.      * Retrieve a previously registered <b>IMediator</b> instance from the <b>View</b>.
  292.      *
  293.      * @param string $mediatorName Name of the <b>IMediator</b> instance to retrieve.
  294.      * @return IMediator The <b>IMediator</b> previously registered with the given <var>mediatorName</var>.
  295.      */
  296.     public function retrieveMediator( $mediatorName )
  297.     {
  298.         return ( $this->hasMediator( $mediatorName ) ? $this->mediatorMap[ $mediatorName ] : null );
  299.     }
  300.  
  301.     /**
  302.      * Remove Mediator
  303.      *
  304.      * Remove a previously registered <b>IMediator</b> instance from the <b>View</b>.
  305.      *
  306.      * @param string $mediatorName Name of the <b>IMediator</b> instance to be removed.
  307.      * @return IMediator The <b>IMediator</b> instance previously registered with the given <var>mediatorName</var>.
  308.      */
  309.     public function removeMediator( $mediatorName )
  310.     {
  311.         // Retrieve the named mediator
  312.         $mediator = $this->retrieveMediator( $mediatorName );
  313.  
  314.         if ( $mediator )
  315.         {
  316.             // for every notification this mediator is interested in...
  317.             $interests = $mediator->listNotificationInterests();
  318.             for ( $i = 0; $i < count( $interests ); $i++ )
  319.             {
  320.                 // remove the observer linking the mediator
  321.                 // to the notification interest
  322.                 $this->removeObserver( $interests[$i], $mediator );
  323.             }
  324.  
  325.             // remove the mediator from the map
  326.             unset($this->mediatorMap[ $mediatorName ]);
  327.  
  328.             // alert the mediator that it has been removed
  329.             $mediator->onRemove();
  330.         }
  331.  
  332.         return $mediator;
  333.     }
  334.  
  335.     /**
  336.      * Has Mediator
  337.      *
  338.      * Check if a <b>IMediator</b> is registered or not.
  339.      *
  340.      * @param string $mediatorName The name of the <b>IMediator</b> to check for.
  341.      * @return bool Boolean: Whether a <b>IMediator</b> is registered with the given <var>mediatorName</var>.
  342.      */
  343.     public function hasMediator( $mediatorName )
  344.     {
  345.         return isset( $this->mediatorMap[ $mediatorName ] );
  346.     }
  347.  
  348.     /**
  349.      * Remove View
  350.      *
  351.      * Remove an <b>IView</b> instance by key.
  352.      *
  353.      * @param string $key The multitonKey of IView instance to remove
  354.      * @return void 
  355.      */
  356.     public static function removeView( $key )
  357.     {
  358.         unset( self::$instanceMap[ $key ] );
  359.     }
