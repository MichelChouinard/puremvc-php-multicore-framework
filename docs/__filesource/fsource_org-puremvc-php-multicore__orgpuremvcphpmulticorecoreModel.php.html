  24. /**
  25.  * A Multiton <b>IModel</b> implementation.
  26.  *
  27.  * In PureMVC, the <b>Model</b> class provides
  28.  * access to model objects (Proxies) by named lookup.
  29.  *
  30.  * The <b>Model</b> assumes these responsibilities:
  31.  *
  32.  * - Maintain a cache of <b>IProxy</b> instances.

  34.  *   <b>IProxy</b> instances.
  35.  *
  36.  * Your application must register <b>IProxy</b> instances

  38.  * <b>ICommand</b> to create and register <b>IProxy</b>

  40.  * actors.
  41.  
  42.  * @see Proxy
  43.         org\puremvc\php\multicore\patterns\proxy\Proxy.php
  44.  * @see IProxy
  45.         org\puremvc\php\multicore\interfaces\IProxy.php
  46.  *
  47.  * @package org.puremvc.php.multicore
  48.  */
  49. class Model implements IModel

  57.     /**

  61.     protected $proxyMap = array();

  63.     /**

  74.  
  75.     /**
  76.      * Instance constructor
  77.      *
  78.      * This <b>IModel</b> implementation is a Multiton, so you should not
  79.      * call the constructor directly, but instead call the static Multiton
  80.      * Factory method.
  81.      *
  82.      * ex:
  83.      * <code>
  84.      * Model::getInstance( 'multitonKey' )
  85.      * </code>
  86.      *
  87.      * @param string $key Unique key for this instance.
  88.      * @throws Exception if instance for this key has already been constructed
  89.      * @return IModel Multiton instance for this key
  90.      */
  91.     protected function __construct( $key )
  92.     {
  93.         if ( isset( self::$instanceMap[ $key ] ) )
  94.         {

  96.         }
  97.         $this->multitonKey = $key;
  98.         self::$instanceMap[ $this->multitonKey ] = $this;
  99.         $this->proxyMap = array();

  101.     }
  102.

  108.      * the constructor.
  109.      *
  110.      * @return void 
  111.      */
  112.     protected function initializeModel(  )
  113.     {
  114.     }
  115.  
  116.     /**
  117.      * Model Factory method.
  118.      *
  119.      * This <b>IModel</b> implementation is a Multiton so
  120.      * this method MUST be used to get acces, or create, <b>IModel</b>s.
  121.      *
  122.      * @param string $key Unique key for this instance.

  125.     public static function getInstance( $key )

  127.         if ( !isset( self::$instanceMap[ $key ] ) )

  129.             self::$instanceMap[$key] = new Model( $key );
  130.         }
  131.

  135.     /**

  140.      * @param IProxy $proxy The <b>IProxy</b> to be registered with the <b>Model</b>.

  143.     public function registerProxy(IProxy $proxy )

  146.         $this->proxyMap[ $proxy->getProxyName() ] = $proxy;

  158.     public function retrieveProxy( $proxyName )

  161.     }
  162.  
  163.     /**

  171.     public function hasProxy( $proxyName )
  172.     {
  173.         return isset( $this->proxyMap[ $proxyName ] );
  174.     }
  175.  
  176.     /**
  177.      * Remove Proxy
  178.      *
  179.      * Remove a previously registered <b>IProxy</b> instance from the <b>Model</b> by name.

  181.      * @param string $proxyName Name of the <b>IProxy</b> to remove from the <b>Model</b>.

  184.     public function removeProxy( $proxyName )
  185.     {
  186.         $proxy = $this->retrieveProxy( $proxyName );
  187.         if ($proxy )
  188.         {
  189.             unset( $this->proxyMap[ $proxyName ] );
  190.             $proxy->onRemove();
  191.         }
  192.         return $proxy;
  193.     }
  194.  
  195.     /**

  199.      *
  200.      * @param string $key The multitonKey of IModel instance to remove
  201.      * @return void 
  202.      */
  203.     public static function removeModel( $key )
  204.     {
  205.         unset( self::$instanceMap[ $key ] );
  206.     }

  208. }
