  27. /**
  28.  * The interface definition for a PureMVC Facade.
  29.  *
  30.  * The <b>Facade</b> Pattern suggests providing a single
  31.  * class to act as a central point of communication
  32.  * for a subsystem.
  33.  *
  34.  * In PureMVC, the <b>Facade</b> acts as an interface between
  35.  * the core MVC actors (Model, View, Controller) and
  36.  * the rest of your application.
  37.  *
  38.  * @see IModel
  39.         org\puremvc\php\multicore\interfaces\IModel.php
  40.  * @see IView
  41.         org\puremvc\php\multicore\interfaces\IView.php
  42.  * @see IController
  43.         org\puremvc\php\multicore\interfaces\IController.php
  44.  * @see ICommand
  45.         org\puremvc\php\multicore\interfaces\ICommand.php
  46.  * @see INotification
  47.         org\puremvc\php\multicore\interfaces\INotification.php
  48.  * @package org.puremvc.php.multicore
  49.  *
  50.  */
  51. interface IFacade extends INotifier
  52. {
  53.  
  54.     /**
  55.      * Register Proxy
  56.      *
  57.      * Register an <b>IProxy</b> with the <b>Model</b>.
  58.      *
  59.      * @param IProxy $proxy The <b>IProxy</b> to be registered with the <b>Model</b>.
  60.      * @return void 
  61.      */
  62.     public function registerProxy( IProxy $proxy );
  63.  
  64.     /**
  65.      * Retreive Proxy
  66.      *
  67.      * Retrieve a previously registered <b>IProxy</b> from the <b>Model</b> by name.
  68.      *
  69.      * @param string $proxyName Name of the <b>IProxy</b> instance to be retrieved.
  70.      * @return IProxy The <b>IProxy</b> previously regisetered by <var>proxyName</var> with the <b>Model</b>.
  71.      */
  72.     public function retrieveProxy( $proxyName );
  73.  
  74.     /**
  75.      * Remove Proxy
  76.      *
  77.      * Remove a previously registered <b>IProxy</b> instance from the <b>Model</b> by name.
  78.      *
  79.      * @param string $proxyName Name of the <b>IProxy</b> to remove from the <b>Model</b>.
  80.      * @return IProxy The <b>IProxy</b> that was removed from the <b>Model</b>.
  81.      */
  82.     public function removeProxy( $proxyName );
  83.  
  84.     /**
  85.      * Has Proxy
  86.      *
  87.      * Check if a Proxy is registered for the given <var>proxyName</var>.
  88.      *
  89.      * @param string $proxyName Name of the <b>Proxy</b> to check for.
  90.      * @return bool Boolean: Whether a <b>Proxy</b> is currently registered with the given <var>proxyName</var>.
  91.      */
  92.     public function hasProxy( $proxyName );
  93.  
  94.     /**
  95.      * Register Command
  96.      *
  97.      * Register an <b>ICommand</b> with the <b>Controller</b>.
  98.      *
  99.      * @param string $noteName Name of the <b>INotification</b>
  100.      * @param object|string$commandClassRef <b>ICommand</b> object to register. Can be an object OR a class name.
  101.      * @return void 
  102.      */
  103.     public function registerCommand( $noteName, $commandClassRef );
  104.  
  105.     /**
  106.      * Remove Command
  107.      *
  108.      * Remove a previously registered <b>ICommand</b> to <b>INotification</b> mapping.
  109.      *
  110.      * @param string $notificationName Name of the <b>INotification</b> to remove the <b>ICommand</b> mapping for.
  111.      */
  112.     public function removeCommand( $notificationName );
  113.  
  114.     /**
  115.      * Has Command

  117.      * Check if a <b>Command</b> is registered for a given <b>Notification</b>
  118.      *
  119.      * @param string $notificationName Name of the <b>INotification</b> to check for.
  120.      * @return bool Whether a <b>Command</b> is currently registered for the given <var>notificationName</var>.
  121.      */
  122.     public function hasCommand( $notificationName );
  123.  
  124.     /**
  125.      * Register Mediator
  126.      *
  127.      * Register an <b>IMediator</b> instance with the <b>View</b>.
  128.      *
  129.      * @param IMediator $mediator Reference to the <b>IMediator</b> instance.
  130.      */
  131.     public function registerMediator( IMediator $mediator );
  132.  
  133.     /**
  134.      * Retreive Mediator
  135.      *
  136.      * Retrieve a previously registered <b>IMediator</b> instance from the <b>View</b>.
  137.      *
  138.      * @param string $mediatorName Name of the <b>IMediator</b> instance to retrieve.
  139.      * @return IMediator The <b>IMediator</b> previously registered with the given <var>mediatorName</var>.
  140.      */
  141.     public function retrieveMediator( $mediatorName );
  142.  
  143.     /**
  144.      * Remove Mediator

  146.      * Remove a previously registered <b>IMediator</b> instance from the <b>View</b>.
  147.      *
  148.      * @param string $mediatorName Name of the <b>IMediator</b> instance to be removed.
  149.      * @return IMediator The <b>IMediator</b> instance previously registered with the given <var>mediatorName</var>.
  150.      */
  151.     public function removeMediator( $mediatorName );
  152.  
  153.     /**
  154.      * Has Mediator
  155.      *
  156.      * Check if a <b>IMediator</b> is registered or not.
  157.      *
  158.      * @param string $mediatorName The name of the <b>IMediator</b> to check for.
  159.      * @return bool Boolean: Whether a <b>IMediator</b> is registered with the given <var>mediatorName</var>.
  160.      */
  161.     public function hasMediator( $mediatorName );
  162.  
  163.     /**
  164.      * Notify <b>Observer</b>s.
  165.      *
  166.      * This method is left public mostly for backward
  167.      * compatibility, and to allow you to send custom
  168.      * notification classes using the facade.
  169.      *
  170.      * Usually you should just call sendNotification
  171.      * and pass the parameters, never having to
  172.      * construct the notification yourself.
  173.      *
  174.      * @param INotification $notification The <b>INotification</b> to have the <b>View</b> notify <b>Observers</b> of.
  175.      * @return void 
  176.      */
  177.     public function notifyObservers( INotification $notification );
  178.  
  179. }
